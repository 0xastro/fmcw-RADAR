752 int32_t symMatInv(float * restrict inv, float const * restrict m){
753  float det, invDet;
754  int32_t ik;
755 #if KF_2D
756  /* The 2D Kalman filter's residual covariance matrix is a 3x3 matrix */
757  /* Since the Adjoint32_t is also symmetric, we only need to compute 6 */
758  /* components of the adjoint. */
759  inv[0] = (m[3] * m[5]) - (m[4] * m[4]);
760  inv[1] = (m[4] * m[2]) - (m[1] * m[5]);
761  inv[2] = (m[1] * m[4]) - (m[3] * m[2]);
762  inv[3] = (m[0] * m[5]) - (m[2] * m[2]);
763  inv[4] = (m[1] * m[2]) - (m[0] * m[4]);
764  inv[5] = (m[0] * m[3]) - (m[1] * m[1]);
765 
766  det = m[0] * inv[0] + m[1] * inv[1] + m[2] * inv[2];
767 
768 
769 #elif KF_3D
770  /* The 3D Kalman filter's residual covariance matrix is a 4x4 matrix */
771  /* Since the Adjoint is also symmetric, we only need to compute 10 */
772  /* components of the adjoint. */
773  inv[0] = m[4] * (m[7] * m[9] - m[8] * m[8]) - m[5] * (m[5] * m[9] - m[6] * m[8]) + m[6] * (m[5] * m[8] - m[6] * m[7]);
774  inv[1] = -m[1] * (m[7] * m[9] - m[8] * m[8]) + m[2] * (m[5] * m[9] - m[6] * m[8]) - m[3] * (m[5] * m[8] - m[6] * m[7]);
775  inv[2] = m[1] * (m[5] * m[9] - m[6] * m[8]) - m[2] * (m[4] * m[9] - m[6] * m[6]) + m[3] * (m[4] * m[8] - m[5] * m[6]);
776  inv[3] = -m[1] * (m[5] * m[8] - m[6] * m[7]) + m[2] * (m[4] * m[8] - m[5] * m[6]) - m[3] * (m[4] * m[7] - m[5] * m[5]);
777  inv[4] = m[0] * (m[7] * m[9] - m[8] * m[8]) - m[2] * (m[2] * m[9] - m[3] * m[8]) + m[3] * (m[2] * m[8] - m[3] * m[7]);
778  inv[5] = -m[0] * (m[5] * m[9] - m[6] * m[8]) + m[2] * (m[1] * m[9] - m[3] * m[6]) - m[3] * (m[1] * m[8] - m[3] * m[5]);
779  inv[6] = m[0] * (m[5] * m[8] - m[6] * m[7]) - m[2] * (m[1] * m[8] - m[2] * m[6]) + m[3] * (m[1] * m[7] - m[2] * m[5]);
780  inv[7] = m[0] * (m[4] * m[9] - m[6] * m[6]) - m[1] * (m[1] * m[9] - m[3] * m[6]) + m[3] * (m[1] * m[6] - m[3] * m[4]);
781  inv[8] = -m[0] * (m[4] * m[8] - m[5] * m[6]) + m[1] * (m[1] * m[8] - m[2] * m[6]) - m[3] * (m[1] * m[5] - m[2] * m[4]);
782  inv[9] = m[0] * (m[4] * m[7] - m[5] * m[5]) - m[1] * (m[1] * m[7] - m[2] * m[5]) + m[2] * (m[1] * m[5] - m[2] * m[4]);
783  det = m[0] * inv[0] + m[1] * inv[1] + m[2] * inv[2] + m[3] * inv[3];
784 #endif
785 
786  /* Correlation matrices are positive semi-definite. We also
787  * whether it is singular or nan. */
788  if (det > EPS)
789  {
790  invDet = recipsp(det);
791 
792  for (ik = 0; ik < N_UNIQ_ELEM_IN_SYM_RESIDCOVMAT; ik++)
793  {
794  inv[ik] = inv[ik] * invDet;
795  }
796 
797  return true;
798  }
799  else
800  {
801  return false;
802  }
